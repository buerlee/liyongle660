\begin{ti}
	设 $g(x) = \begin{cases}
		2 - x, & x \leq 0 \\
		2 + x, & x > 0
	\end{cases}$, $f(x) = \begin{cases}
		x^{2}, & x < 0 \\
		-x, & x \geq 0
	\end{cases}$，则 $\lim_{x \to 0} g(f(x)) = $ \hua。
\end{ti}

\end{ti}

\end{ti}

\begin{ti}
