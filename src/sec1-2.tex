	\begin{tasks}
		\task 如果 $f(x)$ 在 $x_0$ 的任意空心邻域内无界，则 $\lim_{x \to x_0} f(x) = \infty$。
		\task 如果 $\lim_{x \to x_0} f(x) = \infty$，则 $f(x)$ 在 $x_0$ 的任意空心邻域内无界。

	\end{tasks}
